//Программа для подсчёта суммы всех чисел от 1 до 100 включительно, которые не кратны 3

let total = 0;

for(let i = 1; i <= 100; i++){
  if(i%3 !== 0){
    total = total + i;
  }
}

//Функция для возведения числа в степень
let start, rate, result = 1

function exponention(start, rate){
  for(let i=1;i<=rate;i++){
    result = start * result 
  }
}

//Первый аргумент это число, второй - степень в которую нужно возвести
