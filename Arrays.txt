function createFindFunction(company) {
  function findValueByKey(companyName, company) {
      if (company.name === companyName) {
          // Повертаємо об'єкт без полів partners та clients, якщо вони існують
          const { partners, clients, ...filteredCompany } = company;
          return filteredCompany;
      }
      if (company.clients) {
          for (let client of company.clients) {
              const result = findValueByKey(companyName, client);
              if (result) {
                  return result;
              }
          }
      }
      if (company.partners) {
          for (let partner of company.partners) {
              const result = findValueByKey(companyName, partner);
              if (result) {
                  return result;
              }
          }
      }
      return null;
  }
  return function(companyName) {
      return findValueByKey(companyName, company);
  };
}

// Створюємо функцію пошуку для конкретного об'єкта company
const findValueInCompany = createFindFunction(company);

// Приклад виклику функції
const companyName = 'Клієнт 1.2.3';//Сюди вводити назву компанії, яку шукаєте
