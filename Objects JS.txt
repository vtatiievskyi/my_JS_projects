let sum = 0, transient, transient2, transient3 = 0, min = Infinity, max = 0

var services = {
	"стрижка": "60 грн",
	"гоління": "80 грн",
	"Миття голови": "100 грн",
  price(){
      for (let i in services){
      if(typeof services[i] === "function"){
        continue
      } 
      transient = services[i].substring(0, services[i].length - 4)
      sum = sum + Number(transient)
      
  }
  console.log("Загальна вартість наданих послуг: " + sum + " грн")
  sum = 0
},
minPrice(){
  for (let i in services){
    if(typeof services[i] === "function"){
      continue
    } 
  transient2 = services[i].substring(0, services[i].length - 4)
  if (Number(transient2) < min){
    min = transient2
  }}
  console.log("Мінімальна вартість послуги: " + min + " грн")
},
maxPrice(){
  for (let i in services){
    if(typeof services[i] === "function"){
      continue
    } 
  transient3 = services[i].substring(0, services[i].length - 4)
  if (Number(transient3) > max){
    max = transient3
  }}
  console.log("Максимальна вартість послуги: " + max + " грн")
}
}
